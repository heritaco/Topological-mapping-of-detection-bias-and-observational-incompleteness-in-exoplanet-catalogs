\section{Business Understanding}

\begin{frame}[plain]
  \phasebox{Fase 1}{Business Understanding}{Definir el problema, el alcance y lo que no se debe afirmar}
\end{frame}

\begin{frame}{Problema: los exoplanetas no forman cajitas limpias}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \item El cat\'alogo mezcla planetas peque\~nos, sub-Neptunos, gigantes y \'orbitas muy distintas.
        \item Muchas transiciones son graduales, no separaciones perfectas.
        \item La observaci\'on tambi\'en deja huella: no todos los planetas son igual de detectables.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{alertblock}{Riesgo}
        Una regi\'on visible puede reflejar f\'isica, sesgo observacional, imputaci\'on o una mezcla de las tres.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Pregunta de investigaci\'on}
  \begin{block}{Pregunta}
    ?`Puede Mapper revelar regiones estructuradas en variables f\'isicas, orbitales y t\'ermicas, y ayudar a distinguir se\~nal astrof\'isica plausible de sesgo observacional e incertidumbre por imputaci\'on?
  \end{block}

  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Buscamos}
        \begin{itemize}
          \item regiones candidatas;
          \item patrones topol\'ogicos;
          \item evidencia medible para priorizar inspecci\'on.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{No buscamos}
        \begin{itemize}
          \item clases definitivas;
          \item taxonom\'ia final de planetas;
          \item probar la topolog\'ia real del universo.
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Data Understanding}

\begin{frame}[plain]
  \phasebox{Fase 2}{Data Understanding}{Entender el cat\'alogo, sus variables, sus faltantes y sus sesgos}
\end{frame}

\begin{frame}{Dataset usado}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item Fuente: NASA Exoplanet Archive.
        \item Tabla: \texttt{PSCompPars}.
        \item Tama\~no analizado: 6273 exoplanetas.
        \item Unidad de an\'alisis: un planeta como punto en un espacio de variables.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Idea central}
        El dataset es suficientemente rico para explorar estructura, pero no est\'a listo para Mapper sin preparaci\'on.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Variables disponibles vs. variables \'utiles}
  \begin{block}{Decisi\'on}
    No usamos todas las columnas como coordenadas de Mapper.
  \end{block}

  \begin{itemize}
    \item Algunas columnas son identificadores, referencias, incertidumbres o metadata observacional.
    \item Usarlas todas puede crear geometr\'ia dominada por ruido, redundancia o proceso de descubrimiento.
    \item Usamos vistas peque\~nas e interpretables: f\'isica, orbital, t\'ermica y conjunta.
  \end{itemize}

  \takeaway{Menos variables no significa menos rigor; significa m\'as control interpretativo.}
\end{frame}

\begin{frame}{Missingness: el cat\'alogo no est\'a completo}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item Mapper requiere una matriz num\'erica completa.
        \item Eliminar filas incompletas reducir\'ia demasiado el cat\'alogo.
        \item La soluci\'on fue imputar y despu\'es auditar la imputaci\'on.
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \fig[0.98\linewidth]{01_missingness_before_after.pdf}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Sesgo observacional como parte del dato}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item \texttt{discoverymethod}: c\'omo fue descubierto.
        \item \texttt{disc\_year}: cu\'ando fue descubierto.
        \item \texttt{disc\_facility}: con qu\'e instalaci\'on o misi\'on.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Uso correcto}
        Estas variables no construyen el Mapper. Se usan despu\'es para preguntar si las regiones est\'an alineadas con el proceso observacional.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Data Preparation}

\begin{frame}[plain]
  \phasebox{Fase 3}{Data Preparation}{Construir una matriz completa, f\'isicamente razonable y auditable}
\end{frame}

\begin{frame}{Imputaci\'on principal}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item M\'etodo principal: \texttt{iterative}.
        \item La imputaci\'on permite conservar cobertura del cat\'alogo.
        \item No se oculta: se registra por planeta, nodo y grafo.
      \end{itemize}
      \begin{alertblock}{Regla de interpretaci\'on}
        Alta imputaci\'on reduce confianza, aunque el grafo sea complejo.
      \end{alertblock}
    \end{column}
    \begin{column}{0.52\textwidth}
      \fig[0.98\linewidth]{02_value_source_composition.pdf}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Derivaciones f\'isicas}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{\texttt{pl\_dens}}
        Densidad planetaria derivada a partir de masa y radio cuando fue posible.
      \end{block}
      \begin{block}{\texttt{pl\_orbsmax}}
        Semieje mayor derivado desde periodo orbital y masa estelar cuando fue posible.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Lectura}
        Derivado no significa observado directamente. Por eso \texttt{pl\_dens} se trata como control cuando aparece junto a masa y radio.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Variables de construcci\'on vs. auditor\'ia}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \begin{block}{Construyen la geometr\'ia}
        \small
        \texttt{pl\_rade}, \texttt{pl\_bmasse}, \texttt{pl\_dens}, \texttt{pl\_orbper}, \texttt{pl\_orbsmax}, \texttt{pl\_insol}, \texttt{pl\_eqt}.
      \end{block}
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{alertblock}{Auditan la geometr\'ia}
        \small
        \texttt{discoverymethod}, \texttt{disc\_year}, \texttt{disc\_facility}.
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \takeaway{Primero construimos regiones con variables f\'isicas/orbitales/t\'ermicas; despu\'es preguntamos si esas regiones parecen sesgadas.}
\end{frame}

\begin{frame}{Espacios de variables}
  \small
  \begin{tabular}{lll}
    \toprule
    Espacio & Variables & Lectura \\
    \midrule
    \texttt{phys\_min} & \texttt{pl\_rade}, \texttt{pl\_bmasse} & qu\'e es el planeta \\
    \texttt{phys\_density} & + \texttt{pl\_dens} & control con densidad \\
    \texttt{orbital} & \texttt{pl\_orbper}, \texttt{pl\_orbsmax} & d\'onde vive \\
    \texttt{thermal} & \texttt{pl\_insol}, \texttt{pl\_eqt} & ambiente energ\'etico \\
    \texttt{orb\_thermal} & orbital + t\'ermico & \'orbita y energ\'ia \\
    \texttt{joint\_no\_density} & f\'isico + orbital + t\'ermico & vista conjunta sin densidad \\
    \texttt{joint} & joint + \texttt{pl\_dens} & control conjunto con densidad \\
    \bottomrule
  \end{tabular}

  \vspace{0.7em}
  \begin{alertblock}{Alcance actual}
    El espacio orbital actual no incluye \texttt{pl\_orbeccen}; la excentricidad queda como extensi\'on futura.
  \end{alertblock}
\end{frame}

\section{Modeling}

\begin{frame}[plain]
  \phasebox{Fase 4}{Modeling}{Construir grafos Mapper reproducibles a partir de las vistas preparadas}
\end{frame}

\begin{frame}{Mapper en una diapositiva}
  \begin{enumerate}
    \item Elegir un espacio de variables.
    \item Calcular una lente: principal = \texttt{pca2}.
    \item Cubrir la lente con celdas solapadas.
    \item Clusterizar localmente dentro de cada celda con DBSCAN.
    \item Convertir cada cluster local en nodo.
    \item Conectar nodos que comparten planetas.
  \end{enumerate}

  \takeaway{Mapper produce un grafo de regiones solapadas, no una partici\'on r\'igida del cat\'alogo.}
\end{frame}

\begin{frame}{Qu\'e significa el grafo Mapper}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.49\textwidth}
      \begin{block}{Nodo}
        Regi\'on local de planetas parecidos.
      \end{block}
      \begin{block}{Arista}
        Dos nodos comparten al menos un planeta.
      \end{block}
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{block}{Componente}
        Zona conectada del mapa.
      \end{block}
      \begin{block}{Ciclo}
        Complejidad topol\'ogica:
        \[
          \beta_1 = |E| - |V| + \beta_0.
        \]
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Por qu\'e DBSCAN local}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\textwidth}
      \begin{itemize}
        \item No fija de antemano el n\'umero de clusters.
        \item Funciona bien como clusterizador local dentro de cada celda.
        \item Permite regiones no perfectamente esf\'ericas.
        \item Puede dejar puntos aislados como ruido local.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Configuraci\'on usada}
        \small
        \begin{tabular}{ll}
          \toprule
          Par\'ametro & Valor \\
          \midrule
          \texttt{clusterer} & \texttt{dbscan} \\
          \texttt{min\_samples} & 4 \\
          \texttt{eps\_percentile} & 90.0 \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Lentes y cover}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{Lente principal}
        \texttt{pca2}: capa principal de interpretaci\'on.
      \end{block}
      \begin{block}{Sensibilidad}
        \texttt{domain} y \texttt{density}: lentes para comparar estabilidad.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Cover fijo}
        \texttt{n\_cubes=10}, \texttt{overlap=0.35}, es decir \texttt{cubes10\_overlap0p35}.
      \end{alertblock}
      \small No se reporta como full grid search de cover.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Modelos generados}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item 7 espacios de variables.
        \item 3 lentes.
        \item 21 grafos Mapper.
        \item Cada grafo tiene JSON, nodos, aristas y configuraci\'on.
      \end{itemize}
    \end{column}
    \begin{column}{0.52\textwidth}
      \fig[0.96\linewidth]{07_mapper_lens_sensitivity.pdf}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Mapa final de evidencia}

\begin{frame}{Categor\'ias finales}
  \small
  \begin{tabular}{ll}
    \toprule
    Categor\'ia & Interpretaci\'on \\
    \midrule
    \texttt{physical} & coherencia f\'isica/orbital, baja imputaci\'on y bajo sesgo \\

    \texttt{weak} & alta imputaci\'on, regi\'on peque\~na o evidencia insuficiente \\
    \bottomrule
  \end{tabular}

  \vspace{0.8em}
  \begin{block}{Idea}
    Estas son categor\'ias de evidencia, no clases planetarias.
  \end{block}
\end{frame}

\begin{frame}{Qu\'e s\'i afirmamos / qu\'e no afirmamos}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \begin{block}{S\'i afirmamos}
        \begin{itemize}
          \item Mapper encontr\'o regiones estructuradas.
          \item El orbital fue el candidato m\'as informativo.
          \item La imputaci\'on y el sesgo cambian la interpretaci\'on.
          \item Hay regiones para inspecci\'on posterior.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{No afirmamos}
        \begin{itemize}
          \item Clases definitivas de exoplanetas.
          \item Topolog\'ia real del universo.
          \item Causalidad de \texttt{discoverymethod}.
          \item Que m\'as variables siempre sea mejor.
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Limitaciones y trabajo futuro}
  \begin{itemize}
    \item El cover reconciliado es fijo: \texttt{cubes10\_overlap0p35}.
    \item El espacio orbital actual no incluye excentricidad.
    \item La prueba de permutaci\'on mide asociaci\'on, no causalidad.
    \item Las regiones \texttt{weak} no deben sobreinterpretarse.
    \item Futuro: sensibilidad de cover, excentricidad orbital, validaci\'on astrof\'isica externa y revisi\'on por regiones.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusi\'on final}
  \begin{block}{Conclusi\'on defendible}
    Mapper/TDA es \'util para priorizar regiones de inspecci\'on en el cat\'alogo de exoplanetas, siempre que sus grafos se lean junto con imputaci\'on, sesgo observacional y sensibilidad metodol\'ogica.
  \end{block}

  \begin{itemize}
    \item El resultado m\'as fuerte es orbital: complejo y con baja imputaci\'on.
    \item La auditor\'ia muestra asociaci\'on con \texttt{discoverymethod}.
    \item La conclusi\'on honesta es un mapa de evidencia, no una taxonom\'ia final.
  \end{itemize}
\end{frame}
